def clean(self):
    pwd = self.cleaned_data.get('password')
    re_pwd = self.cleaned_data.get('re_password')
    if pwd == re_pwd:
        return self.cleaned_data
    self.add_error('re_password', '两次密码不一致')  # 给字段单独加上错误信息
    raise ValidationError('格式错误')

使用局部钩子的代码示例：

def clean_re_password(self):

    re_pwd = self.cleaned_data.get('re_password')
    if pwd == re_pwd:
        return re_pwd
    raise ValidationError('格式错误')
